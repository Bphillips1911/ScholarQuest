// ENHANCED: API route for worksheet generation with passages and visuals
// Location: app/api/admin/worksheet/generate/route.js

import { NextResponse } from 'next/server';

export async function POST(request) {
  try {
    const { standard, dokLevel, itemCount, subject, grade } = await request.json();

    if (!standard || !dokLevel || !itemCount) {
      return NextResponse.json(
        { error: 'Missing required fields' },
        { status: 400 }
      );
    }

    const isELA = subject === 'ELA' || subject === 'English Language Arts';
    const isMath = subject === 'Math' || subject === 'Mathematics';

    console.log('Generating enhanced worksheet with visuals...');

    const prompt = buildEnhancedPrompt(subject, grade, standard, dokLevel, itemCount, isELA, isMath);

    // Use Gemini 1.5 Flash for better results
    const geminiResponse = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${process.env.GEMINI_API_KEY}`,
      {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          contents: [{
            parts: [{
              text: prompt
            }]
          }],
          generationConfig: {
            temperature: 0.8,
            topK: 40,
            topP: 0.95,
            maxOutputTokens: 8192,
          }
        })
      }
    );

    if (!geminiResponse.ok) {
      const errorData = await geminiResponse.json();
      console.error('Gemini API Error:', errorData);
      
      if (geminiResponse.status === 429) {
        return NextResponse.json(
          { 
            error: 'Rate limit reached. Please wait a moment and try again.',
            retryAfter: 60
          },
          { status: 429 }
        );
      }
      
      throw new Error(errorData.error?.message || 'Gemini API request failed');
    }

    const geminiData = await geminiResponse.json();
    
    if (!geminiData.candidates || !geminiData.candidates[0]) {
      throw new Error('No response from Gemini AI');
    }

    const generatedText = geminiData.candidates[0].content.parts[0].text;
    
    // Clean up response
    let cleanedText = generatedText.trim();
    cleanedText = cleanedText.replace(/```json\s*/g, '');
    cleanedText = cleanedText.replace(/```\s*/g, '');
    
    // Extract JSON
    const jsonMatch = cleanedText.match(/\{[\s\S]*\}/);
    if (!jsonMatch) {
      console.error('Failed to parse JSON from:', generatedText.substring(0, 500));
      throw new Error('Could not extract valid JSON from AI response');
    }
    
    const worksheet = JSON.parse(jsonMatch[0]);

    // Validate worksheet structure
    if (!worksheet.questions || !Array.isArray(worksheet.questions)) {
      throw new Error('Invalid worksheet structure');
    }

    console.log(`Successfully generated ${worksheet.questions.length} questions with ${isELA ? 'passage' : 'visuals'}`);

    return NextResponse.json({
      success: true,
      worksheet: worksheet
    });

  } catch (error) {
    console.error('Error generating worksheet:', error);
    return NextResponse.json(
      { 
        error: 'Failed to generate worksheet', 
        details: error.message 
      },
      { status: 500 }
    );
  }
}

function getDokGuidance(level) {
  const guidance = {
    2: `DOK 2 - Skills and Concepts:
- Apply concepts and procedures
- Make observations and inferences
- Organize and compare data
- Use models and representations
- Interpret information from graphs/charts`,
    
    3: `DOK 3 - Strategic Thinking:
- Analyze patterns and relationships
- Draw conclusions from evidence
- Justify reasoning with multiple sources
- Solve non-routine multi-step problems
- Support claims with evidence`,
    
    4: `DOK 4 - Extended Thinking:
- Synthesize information across contexts
- Critique and evaluate arguments
- Design solutions to complex problems
- Connect concepts across disciplines
- Create original analysis or interpretation`
  };
  
  return guidance[level] || guidance[2];
}
