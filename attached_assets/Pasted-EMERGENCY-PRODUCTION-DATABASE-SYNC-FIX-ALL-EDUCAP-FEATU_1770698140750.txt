app.get('/api/educap/standards', async (req, res) => {
  const { grade, subject } = req.query;
  
  const standards = await db.query.standards.findMany({
    where: and(
      grade ? eq(standards.grade, parseInt(grade)) : undefined,
      subject ? eq(standards.subject, subject) : undefined,
      eq(standards.status, 'active'),
      or(
        isNull(standards.schoolId),  // ← Global standards
        eq(standards.schoolId, req.user?.schoolId)
      )
    )
  });
  
  console.log(`[EduCAP] GET /api/educap/standards | grade=${grade} subject=${subject} | results=${standards.length}`);
  
  res.json(standards);
});
