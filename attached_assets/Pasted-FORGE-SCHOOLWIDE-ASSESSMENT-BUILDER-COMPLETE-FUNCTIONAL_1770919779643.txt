app.post('/api/educap/forge/assessments/:id/differentiate', async (req, res) => {
  try {
    const { id } = req.params;
    const { versionCount = 3 } = req.body;
    
    console.log(`[Forge Differentiate] Creating ${versionCount} versions for assessment ${id}`);
    
    // Get assessment with items
    const assessment = await db.query.forgeAssessments.findFirst({
      where: eq(forgeAssessments.id, parseInt(id)),
      with: { items: true }
    });
    
    if (!assessment || !assessment.items || assessment.items.length === 0) {
      return res.status(400).json({ error: 'Assessment has no items to differentiate' });
    }
    
    const versions = [];
    const versionCodes = ['A', 'B', 'C', 'D', 'E', 'F'];
    
    for (let i = 0; i < versionCount; i++) {
      // Shuffle item order
      const shuffledItems = [...assessment.items].sort(() => Math.random() - 0.5);
      const itemOrder = shuffledItems.map(item => item.id);
      
      // Shuffle option order for each item
      const optionOrder = {};
      shuffledItems.forEach(item => {
        if (item.itemType === 'multiple_choice') {
          const options = JSON.parse(item.options);
          const shuffled = options.sort(() => Math.random() - 0.5);
          optionOrder[item.id] = shuffled.map((_, idx) => idx);
        }
      });
      
      // Save version to database
      const version = await db.insert(differentiatedVersions).values({
        forgeAssessmentId: parseInt(id),
        versionCode: versionCodes[i],
        versionName: `Version ${versionCodes[i]}`,
        itemOrder: JSON.stringify(itemOrder),
        optionOrder: JSON.stringify(optionOrder),
        createdAt: new Date()
      }).returning();
      
      versions.push({
        id: version[0].id,
        code: versionCodes[i],
        name: `Version ${versionCodes[i]}`,
        itemCount: shuffledItems.length
      });
    }
    
    console.log(`[Forge Differentiate] Created ${versions.length} versions`);
    
    res.json({
      success: true,
      versions,
      message: `Created ${versions.length} differentiated versions`
    });
    
  } catch (error) {
    console.error('[Forge Differentiate] Error:', error);
    res.status(500).json({ error: error.message });
  }
});

app.post('/api/educap/forge/sources/:id/parse', async (req, res) => {
  try {
    const { id } = req.params;
    
    console.log('[Forge Parse] Parsing source:', id);
    
    // Get source
    const source = await db.query.forgeSources.findFirst({
      where: eq(forgeSources.id, parseInt(id))
    });
    
    if (!source) {
      return res.status(404).json({ error: 'Source not found' });
    }
    
    console.log('[Forge Parse] Source found:', source.filename);
    
    // Update status to parsing
    await db.update(forgeSources)
      .set({ status: 'parsing' })
      .where(eq(forgeSources.id, parseInt(id)));
    
    // Run parse logic (simplified - extracts items from PDF/DOCX)
    const extractedItems = await parseSourceFile(source);
    
    console.log('[Forge Parse] Extracted items:', extractedItems.length);
    
    // Save extracted items to staging
    const stagedItems = [];
    for (const item of extractedItems) {
      const staged = await db.insert(stagedItems).values({
        forgeSourceId: source.id,
        itemType: item.type,
        stem: item.stem,
        options: JSON.stringify(item.options),
        standardCode: item.suggestedStandard,
        dokLevel: item.suggestedDOK,
        status: 'pending_review'
      }).returning();
      
      stagedItems.push(staged[0]);
    }
    
    // Update source status to parsed
    await db.update(forgeSources)
      .set({ 
        status: 'parsed',
        itemsExtracted: extractedItems.length,
        parsedAt: new Date()
      })
      .where(eq(forgeSources.id, parseInt(id)));
    
    console.log('[Forge Parse] Parse complete');
    
    res.json({
      success: true,
      itemsExtracted: extractedItems.length,
      sourceId: source.id
    });
    
  } catch (error) {
    console.error('[Forge Parse] Error:', error);
    
    // Update source status to error
    await db.update(forgeSources)
      .set({ status: 'error', errorMessage: error.message })
      .where(eq(forgeSources.id, parseInt(req.params.id)));
    
    res.status(500).json({ error: error.message });
  }
});

app.post('/api/educap/schoolwide/generate', async (req, res) => {
  try {
    const { contentArea, grades, itemCount, dokMix, blueprintWeights } = req.body;
    
    console.log('[Schoolwide] Generating assessment:', { contentArea, grades, itemCount });
    
    // Calculate items per DOK level
    const dok2Count = Math.round((itemCount * dokMix.dok2) / 100);
    const dok3Count = Math.round((itemCount * dokMix.dok3) / 100);
    const dok4Count = itemCount - dok2Count - dok3Count;
    
    console.log('[Schoolwide] DOK distribution:', { dok2Count, dok3Count, dok4Count });
    
    // Get available standards for selected grades
    const standards = await db.query.standards.findMany({
      where: and(
        eq(standards.subject, contentArea),
        inArray(standards.grade, grades),
        eq(standards.status, 'active')
      )
    });
    
    console.log('[Schoolwide] Found ${standards.length} standards');
    
    // Generate items using Claude API
    const generatedItems = [];
    const itemsPerDOK = [
      { dok: 2, count: dok2Count },
      { dok: 3, count: dok3Count },
      { dok: 4, count: dok4Count }
    ];
    
    for (const { dok, count } of itemsPerDOK) {
      if (count === 0) continue;
      
      // Select random standards for this DOK level
      const selectedStandards = standards
        .filter(s => s.dokLevels.includes(dok))
        .sort(() => Math.random() - 0.5)
        .slice(0, Math.min(count, standards.length));
      
      // Generate items for each standard
      for (const standard of selectedStandards) {
        const prompt = `Generate 1 ACAP-style ${contentArea} question for:

Standard: ${standard.code}
Description: ${standard.description}
DOK Level: ${dok}
Grade: ${standard.grade}

Requirements:
- Multiple choice with 4 options
- Real-world context
- Realistic distractors based on common errors
- Include correct answer

Return as JSON:
{
  "stem": "question text",
  "options": ["A. ...", "B. ...", "C. ...", "D. ..."],
  "correctAnswer": "B",
  "rationale": "explanation"
}`;

        const response = await fetch('https://api.anthropic.com/v1/messages', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'x-api-key': process.env.ANTHROPIC_API_KEY,
            'anthropic-version': '2023-06-01'
          },
          body: JSON.stringify({
            model: 'claude-sonnet-4-20250514',
            max_tokens: 1000,
            messages: [{ role: 'user', content: prompt }]
          })
        });
        
        const data = await response.json();
        const itemData = JSON.parse(data.content[0].text);
        
        // Save item to database
        const item = await db.insert(items).values({
          standardCode: standard.code,
          itemType: 'multiple_choice',
          stem: itemData.stem,
          options: JSON.stringify(itemData.options),
          correctAnswer: itemData.correctAnswer,
          rationale: itemData.rationale,
          dokLevel: dok,
          status: 'approved',
          generatedBy: 'claude-api',
          createdAt: new Date()
        }).returning();
        
        generatedItems.push(item[0]);
        
        // Stop if we've generated enough
        if (generatedItems.length >= count) break;
      }
    }
    
    console.log('[Schoolwide] Generated ${generatedItems.length} items');
    
    // Create assessment
    const assessment = await db.insert(assessments).values({
      title: `${contentArea} Schoolwide Assessment - ${grades.join(', ')}`,
      subject: contentArea,
      grade: grades[0], // Primary grade
      assessmentType: 'Baseline',
      itemCount: generatedItems.length,
      timeLimitMinutes: generatedItems.length * 1.5, // 1.5 min per item
      status: 'active',
      createdBy: req.user.id,
      generatedVia: 'schoolwide-builder',
      createdAt: new Date()
    }).returning();
    
    // Link items to assessment
    const itemLinks = generatedItems.map((item, index) => ({
      assessmentId: assessment[0].id,
      itemId: item.id,
      orderNumber: index + 1
    }));
    
    await db.insert(assessmentItemsJoin).values(itemLinks);
    
    console.log('[Schoolwide] Assessment created:', assessment[0].id);
    
    res.json({
      success: true,
      assessment: assessment[0],
      itemCount: generatedItems.length
    });
    
  } catch (error) {
    console.error('[Schoolwide] Error:', error);
    res.status(500).json({ error: error.message });
  }
});

app.get('/api/educap/assessments/:id/download', async (req, res) => {
  try {
    const { id } = req.params;
    const { format = 'pdf' } = req.query;
    
    // Get assessment with items
    const assessment = await db.query.assessments.findFirst({
      where: eq(assessments.id, parseInt(id)),
      with: {
        items: {
          with: { item: true }
        }
      }
    });
    
    if (!assessment) {
      return res.status(404).json({ error: 'Assessment not found' });
    }
    
    // Generate PDF or DOCX
    if (format === 'pdf') {
      const pdfBuffer = await generateAssessmentPDF(assessment);
      
      res.setHeader('Content-Type', 'application/pdf');
      res.setHeader('Content-Disposition', `attachment; filename="assessment-${id}.pdf"`);
      res.send(pdfBuffer);
    } else {
      // Generate text format for now
      const text = generateAssessmentText(assessment);
      
      res.setHeader('Content-Type', 'text/plain');
      res.setHeader('Content-Disposition', `attachment; filename="assessment-${id}.txt"`);
      res.send(text);
    }
    
  } catch (error) {
    console.error('[Download] Error:', error);
    res.status(500).json({ error: error.message });
  }
});

function generateAssessmentText(assessment) {
  let text = `${assessment.title}\n`;
  text += `${'='.repeat(60)}\n\n`;
  text += `Subject: ${assessment.subject}\n`;
  text += `Grade: ${assessment.grade}\n`;
  text += `Type: ${assessment.assessmentType}\n`;
  text += `Items: ${assessment.itemCount}\n`;
  text += `Time Limit: ${assessment.timeLimitMinutes} minutes\n\n`;
  text += `${'='.repeat(60)}\n\n`;
  
  assessment.items.forEach((link, index) => {
    const item = link.item;
    text += `QUESTION ${index + 1}\n`;
    text += `Standard: ${item.standardCode} | DOK ${item.dokLevel}\n\n`;
    text += `${item.stem}\n\n`;
    
    if (item.itemType === 'multiple_choice') {
      const options = JSON.parse(item.options);
      options.forEach(opt => {
        text += `${opt}\n`;
      });
    }
    
    text += `\n${'-'.repeat(60)}\n\n`;
  });
  
  return text;
}
