const geminiAI = require('../services/geminiAI');

app.post('/api/educap/schoolwide/generate', async (req, res) => {
  try {
    const { contentArea, grades, itemCount, dokMix } = req.body;
    
    console.log('[Schoolwide] Generating assessment with Gemini:', { 
      contentArea, 
      grades, 
      itemCount 
    });
    
    // Get available standards for selected grades
    const standards = await db.query.standards.findMany({
      where: and(
        eq(standards.subject, contentArea),
        inArray(standards.grade, grades),
        eq(standards.status, 'active')
      )
    });
    
    if (standards.length === 0) {
      return res.status(400).json({ 
        error: 'No standards found for selected criteria' 
      });
    }
    
    console.log(`[Schoolwide] Found ${standards.length} standards`);
    
    // Generate items using Gemini
    const generatedItems = await geminiAI.generateAssessment({
      contentArea,
      grade: grades[0],
      itemCount,
      dokMix,
      standards
    });
    
    console.log(`[Schoolwide] Generated ${generatedItems.length} items`);
    
    // Save items to database
    const savedItems = [];
    for (const itemData of generatedItems) {
      const item = await db.insert(items).values({
        standardCode: itemData.standard?.code || standards[0].code,
        itemType: 'multiple_choice',
        stem: itemData.stem,
        options: JSON.stringify(itemData.options),
        correctAnswer: itemData.correctAnswer,
        rationale: itemData.rationale,
        dokLevel: itemData.dokLevel || 2,
        status: 'approved',
        generatedBy: 'gemini-ai',
        createdAt: new Date()
      }).returning();
      
      savedItems.push(item[0]);
    }
    
    // Create assessment
    const assessment = await db.insert(assessments).values({
      title: `${contentArea} Schoolwide Assessment - Grades ${grades.join(', ')}`,
      subject: contentArea,
      grade: grades[0],
      assessmentType: 'Baseline',
      itemCount: savedItems.length,
      timeLimitMinutes: savedItems.length * 1.5,
      status: 'active',
      createdBy: req.user.id,
      generatedVia: 'schoolwide-builder-gemini',
      createdAt: new Date()
    }).returning();
    
    // Link items to assessment
    const itemLinks = savedItems.map((item, index) => ({
      assessmentId: assessment[0].id,
      itemId: item.id,
      orderNumber: index + 1
    }));
    
    await db.insert(assessmentItemsJoin).values(itemLinks);
    
    console.log('[Schoolwide] Assessment created:', assessment[0].id);
    
    res.json({
      success: true,
      assessment: assessment[0],
      itemCount: savedItems.length,
      message: `Generated ${savedItems.length} items using Google Gemini AI (FREE)`
    });
    
  } catch (error) {
    console.error('[Schoolwide] Error:', error);
    res.status(500).json({ 
      error: error.message,
      details: 'Check server logs for more information'
    });
  }
});
