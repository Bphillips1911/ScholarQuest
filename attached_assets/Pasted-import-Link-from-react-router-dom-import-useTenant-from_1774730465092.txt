import { useTenant } from '../../context/TenantProvider';

  if (!tenant) {
    return (
      <div className="min-h-screen flex items-center justify-center p-8">
        <div className="max-w-xl text-center">
          <h1 className="text-4xl font-bold mb-4">PBIS House of Champions</h1>
          <p className="text-slate-600 mb-6">
            Welcome to the platform. Please access your school using your assigned school subdomain.
          </p>
        </div>
      </div>
    );
  }

  return (

      <header className="bg-white border-b">
        <div className="max-w-6xl mx-auto px-6 py-4 flex items-center gap-4">

            <img src={tenant.logo_url} alt={tenant.school_name} className="h-14 w-14 object-contain" />
          )}

            <p className="text-slate-500">PBIS House of Champions Portal</p>

        </div>
      </header>

        </h2>
        <p className="text-xl text-slate-600 mb-8">
          {tenant.hero_subtitle || 'Track house points, celebrate scholars, and power positive behavior.'}
        </p>

        <div className="flex flex-wrap items-center justify-center gap-4">
          <Link
            to="/login?role=student"

            style={{ backgroundColor: 'var(--tenant-primary)' }}
          >
            Student Login
          </Link>
          <Link
            to="/login?role=teacher"
            className="px-6 py-3 rounded-xl text-slate-900 font-semibold"
            style={{ backgroundColor: 'var(--tenant-secondary)' }}
          >
            Teacher Login
          </Link>
          <Link
            to="/login?role=tenant_admin"
            className="px-6 py-3 rounded-xl text-white font-semibold"
